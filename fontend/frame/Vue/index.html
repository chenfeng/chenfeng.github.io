    # v-model

    
<template>
<div class="addAndMin">
  <div @click="add">+</div>
  <div class="cont">{{step_}}</div>
  <div @click="minus">-</div>
</div> 
      
</template>
    
<script> 
export default {
  name: 'step-size',
  props: {
    step:{
      type: Number,
      default: 50,
    },
    size: {
      type:Number,
      default: 5
    }
  },
  watch: {
    step_ (value) {
      this.$emit('input',value);
    }
  },
  data () {
    return {
      step_:this.step
    };
  },
  methods:{
    add () {
      this.step_ = this.step_ + this.size;
    },
    minus () {
      this.step_ = this.step_ - this.size; 
    }
  }
    
};
</script>
